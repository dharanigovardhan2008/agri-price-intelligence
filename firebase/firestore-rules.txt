rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection
    match /users/{userId} {
      // Allow read for authenticated users
      allow read: if request.auth != null;
      
      // Allow write only for the user's own document
      allow create: if request.auth != null;
      allow update: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Prices collection
    match /prices/{priceId} {
      // Allow anyone to read prices
      allow read: if true;
      
      // Only admins can write prices
      allow write: if request.auth != null && 
                      get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Alerts collection
    match /alerts/{alertId} {
      // Users can only read their own alerts
      allow read: if request.auth != null && 
                     resource.data.userId == request.auth.uid;
      
      // Users can only create/update/delete their own alerts
      allow create: if request.auth != null && 
                       request.resource.data.userId == request.auth.uid;
      allow update: if request.auth != null && 
                       resource.data.userId == request.auth.uid;
      allow delete: if request.auth != null && 
                       resource.data.userId == request.auth.uid;
    }
    
    // Transactions collection
    match /sell_transactions/{transactionId} {
      // Users can only read their own transactions
      allow read: if request.auth != null && 
                     resource.data.userId == request.auth.uid;
      
      // Users can create their own transactions
      allow create: if request.auth != null && 
                       request.resource.data.userId == request.auth.uid;
      
      // Users can update their own transactions
      allow update: if request.auth != null && 
                       resource.data.userId == request.auth.uid;
    }
    
    // Market trends collection
    match /market_trends/{trendId} {
      // Anyone can read trends
      allow read: if true;
      
      // Only system can write (use service account)
      allow write: if false;
    }
    
    // Notifications collection
    match /notifications/{notificationId} {
      // Users can only read their own notifications
      allow read: if request.auth != null && 
                     resource.data.userId == request.auth.uid;
      
      // System creates notifications (service account)
      allow create: if true;
      
      // Users can mark as read
      allow update: if request.auth != null && 
                       resource.data.userId == request.auth.uid &&
                       request.resource.data.diff(resource.data).affectedKeys().hasOnly(['read', 'readAt']);
    }
  }
}
